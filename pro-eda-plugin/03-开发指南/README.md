---

### 插件生命周期

```typescript
// 插件激活时调用
export function activate(api: any) {
	console.log('插件已激活');

	// 注册命令
	api.registerCommand('myCommand', () => {
		// 命令处理逻辑
	});
}

// 插件停用时调用
export function deactivate() {
	console.log('插件已停用');
	// 清理资源
}
```

---

### 权限系统

插件需要在 `eext.json` 中声明权限：

```json
{
	"permissions": {
		"filesystem": true, // 文件系统访问
		"network": true, // 网络请求
		"clipboard": true // 剪贴板访问
	}
}
```
